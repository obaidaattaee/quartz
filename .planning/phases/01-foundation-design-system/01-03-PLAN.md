**1. Create `resources/js/hooks/use-scroll-header.tsx`** (per D-10):

```typescript
import { useEffect, useState } from 'react';

export function useScrollHeader(threshold = 50): boolean {
    const [isScrolled, setIsScrolled] = useState(false);

    useEffect(() => {
        const handleScroll = () => {
            setIsScrolled(window.scrollY > threshold);
        };

        // Check on mount in case page loads scrolled
        handleScroll();

        window.addEventListener('scroll', handleScroll, { passive: true });

        return () => window.removeEventListener('scroll', handleScroll);
    }, [threshold]);

    return isScrolled;
}
```

**3. Create `resources/js/components/theme-toggle.tsx`** (per DSGN-02):

Uses the existing `useAppearance` hook. Cycles through light/dark/system on click. Shows Sun icon for light, Moon for dark, Monitor for system.

```typescript
import { Monitor, Moon, Sun } from 'lucide-react';

import { Button } from '@/components/ui/button';
import { useAppearance } from '@/hooks/use-appearance';
import type { Appearance } from '@/hooks/use-appearance';
import { useLocale } from '@/hooks/use-locale';
import { cn } from '@/lib/utils';

type Props = {
    className?: string;
};

const CYCLE_ORDER: Appearance[] = ['light', 'dark', 'system'];

export default function ThemeToggle({ className }: Props) {
    const { appearance, updateAppearance } = useAppearance();
    const { t } = useLocale();

    const cycleTheme = () => {
        const currentIndex = CYCLE_ORDER.indexOf(appearance);
        const nextIndex = (currentIndex + 1) % CYCLE_ORDER.length;
        updateAppearance(CYCLE_ORDER[nextIndex]);
    };

    const Icon =
        appearance === 'dark' ? Moon : appearance === 'light' ? Sun : Monitor;
    const label =
        appearance === 'dark'
            ? t('theme.dark')
            : appearance === 'light'
              ? t('theme.light')
              : t('theme.system');

    return (
        <Button
            variant="ghost"
            size="icon"
            onClick={cycleTheme}
            className={cn('size-9', className)}
            aria-label={label}
        >
            <Icon className="size-4" />
        </Button>
    );
}
```
